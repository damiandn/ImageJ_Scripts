
var cumulative_x;
var cumulative_y;
var cumulative_z;

macro "shift image down [z]" {
  slicenum = getSliceNumber()

  run("Split Channels");
  selectWindow("C1-Composite");
  setSlice(1)
  run("Add Slice");
  selectWindow("C2-Composite");
  setSlice(nSlices)
  run("Add Slice");
  run("Merge Channels...", "c1=C1-Composite c2=C2-Composite create ignore");
  cumulative_z -= 1;
  print("total z displacement is " + cumulative_z);
  setSlice(slicenum)

}
